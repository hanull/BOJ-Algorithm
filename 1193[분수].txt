import java.util.*;

public class Main {

  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int x = sc.nextInt();
    int i = 0;  
    int sum=0;
    while(sum<x) {
      i++;
      sum+=i;
    }
    int a=0; int b=0; 
    int tmp=i+1; // 분모+분자
    
    if(i%2==0) {
      b = sum-x+1;
      a = tmp-b;
    }else {
      a = sum - x + 1;
      b = tmp - a;
    }
    System.out.println(a+"/"+b);
  }
}
