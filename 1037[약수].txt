import java.util.*;

public class Main {

  public static void main(String[] args){
    Scanner sc = new Scanner(System.in);
    int t = sc.nextInt();
    int[] arr = new int[t];
    for(int i=0; i<t; i++) {
      arr[i] = sc.nextInt();
    }
    Arrays.sort(arr);
    int result = arr[0]*arr[t-1];
    System.out.println(result);
  }
}
