import java.util.*;

public class Main {

  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int[] arr = new int[n];
    for(int i=n; i>0; i--) {
      arr[n-i] = sc.nextInt();
    }
    Arrays.sort(arr);
    for(int i=0; i<arr.length; i++) {
      System.out.println(arr[i]);
    }
  }

}
